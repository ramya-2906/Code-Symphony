class Solution:
    def countMajoritySubarrays(self, nums: List[int], target: int) -> int:
        d=nums
        n=len(nums)
        ans=0
        for i in range(n):
            cnt=0
            for j in range(i,n):
                if nums[j]==target:
                    cnt+=1
                if cnt>(j-i+1)//2:
                    ans+=1
        return ans
