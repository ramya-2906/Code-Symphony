class Solution:
    def lexSmallest(self, s: str) -> str:
        n=len(s)
        be=None
        for k in range(1,n+1):
            a=s[:k][::-1]+s[k:]
            b=s[:-k]+s[-k:][::-1]
            if be is None or a<be:
                be=a
            if b<be:
                be=b
        return be
